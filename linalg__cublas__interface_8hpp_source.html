186  template <typename Matrix,typename Floating> void SpectralBounds(Matrix& H, Floating& Emin, Floating& Emax,const Floating Tolerance)
187  {
188  typedef typename Matrix::value_type Scalar;
189  typedef typename Matrix::index_type Integer;
190  typedef typename Matrix::memory_space MemorySpace;
191 
192  cusparseHandle_t cusparse_handle; //Handle of cusparse
193  cusparseMatDescr_t cusparse_descr; //Cusparse's Matrix Descriptor
194  cusparseHybMat_t GH; //Cusparse's Hybrid Matrix
195  cublasHandle_t cublas_handle; //Handle of cublas
196  cusparseCreate (&cusparse_handle); //Initialization of the cusparse's handler
197  cusparseCreateMatDescr (&cusparse_descr); //Initialization of the descriptor
198  cusparseCreateHybMat (&GH); //Initialization of the Hybrid Matrix
199  cublasCreate (&cublas_handle); //Initialization of the cublas's handler
200  Integer DD=H.num_cols; //Dimensions for the matrix DD*DD
201  cusp::coo_matrix<Integer, Scalar, cusp::host_memory> tempH=H;
202  cusparse::convertX2Hyb(tempH,DD,cusparse_handle,cusparse_descr,GH);
203 
204  /*This method is design to find the spectral bounds of any give simmetric matrix
205  * with at least one eigenvalue passing through zero
206  * we will use the power method in order to find the largest eigenvalue in absolute value
207  * and then translate the eigenvalues to find the largest eigenvalue with oposite sign
208  */
209  cusp::array1d <Scalar, cusp::host_memory> X(DD,0);
210  cusp::array1d <Scalar, cusp::device_memory> GX(DD,0);
211  cusp::array1d <Scalar, cusp::device_memory> GY(DD,0);
212  Scalar* pGX=thrust::raw_pointer_cast(GX.data());
213  Scalar* pGY=thrust::raw_pointer_cast(GY.data());
214  Scalar zero = 0.0f;
215  Scalar one = 1.0f;
216  Scalar diff = -1.0f;
217  /***The method parameters***/
218  Integer it; //Iteration Variable
219  Integer itmax=100; //Iteration max
220  Integer NRandom=10; //Number of test vecto we will try
221  Floating ConvRate; //Convergence Rate Variable
222  Scalar Etemp=0.0; //Temporal Variable for the energy
223  Scalar E0 =0.0; //Temporal Variable for the energy
224  /***We start the iteration cycle, over the test vectors***/
225  for(int r=0;r<NRandom;r++){
226  //Select a random vector
227  for(int i=0;i<DD;i++) X[i]=(Floating)rand();
228  GX =X;
229  //Normalize it
230  cublas::VectorTransform::normalize<Floating>(cublas_handle,DD,pGX);//Here we normalize j0 and then we pass it to jn
231  //Start The convergence cycle//
232  ConvRate=1;
233  it=0;
234  while( (ConvRate>=Tolerance)&& (it<= itmax) ){
235  it=it+1;
236  //We first compute the vector Y=H X
237  cusparse::Multiply(cusparse_handle,cusparse_descr,&one,GH,pGX,&zero,pGY);
238  //Normalize it
239  cublas::VectorTransform::normalize<Floating>(cublas_handle,DD,pGY);
240  //And then we calculate the quantity X=Y-X
241  cublas::VectorTransform::axpy(cublas_handle,DD,&diff,pGY, pGX);
242  //The convergence parameter we choose is the norm |Y-X|
243  cublas::VectorTransform::nrm2(cublas_handle,DD,pGX,&ConvRate);
244  //Notice that because the eigenvalues are defined within a phase
245  //it may happen that Y and X be equal up to a phase of -1
246  //in this case the best thing to do is just take it out
247  ConvRate=abs(ConvRate-2*((int)(ConvRate/(sqrt(2)))));
248  GX=GY;
249  }
251  cusparse::Multiply(cusparse_handle,cusparse_descr,&one,GH,pGX,&zero,pGY);
252  cublas::VectorTransform::dot(cublas_handle,DD,pGX,pGY,&Etemp);
253 
254  //Here we substitute either by the minimal or maximal bound
255  if( Etemp.real()/fabs(Etemp.real()) >0 ) if(Emax < Etemp.real())Emax=Etemp.real();
256  if( Etemp.real()/fabs(Etemp.real()) <0 ) if(Emin > Etemp.real())Emin=Etemp.real();
257  if(fabs(E0.real()) < fabs(Etemp.real())) E0=Etemp;
258  }
259 
260  //Now we shift the hamiltonian
261  for(int k=0;k<tempH.num_entries;k++)
262  if(tempH.row_indices[k]==tempH.column_indices[k])
263  tempH.values[k]=tempH.values[k]-E0;
264  cusparse::convertX2Hyb(tempH,DD,cusparse_handle,cusparse_descr,GH);
265 
266  //Start the Iterations again
267  for(int r=0;r<NRandom;r++){
268  //Select a random vector
269  for(int i=0;i<DD;i++) X[i]=(Floating)rand();
270  GX =X;
271  //Normalize it
272  cublas::VectorTransform::normalize<Floating>(cublas_handle,DD,pGX);//Here we normalize j0 and then we pass it to jn
273  //Start The convergence cycle//
274  ConvRate=1;
275  it=0;
276  while( (ConvRate>=Tolerance)&& (it<= itmax) ){
277  it=it+1;
278  //We first compute the vector Y=H X
279  cusparse::Multiply(cusparse_handle,cusparse_descr,&one,GH,pGX,&zero,pGY);
280  //Normalize it
281  cublas::VectorTransform::normalize<Floating>(cublas_handle,DD,pGY);
282  //And then we calculate the quantity X=Y-X
283  cublas::VectorTransform::axpy(cublas_handle,DD,&diff,pGY, pGX);
284  //The convergence parameter we choose is the norm |Y-X|
285  cublas::VectorTransform::nrm2(cublas_handle,DD,pGX,&ConvRate);
286  //Notice that because the eigenvalues are defined within a phase
287  //it may happen that Y and X be equal up to a phase of -1
288  //in this case the best thing to do is just take it out
289  ConvRate=abs(ConvRate-2*((int)(ConvRate/(sqrt(2)))));
290  GX=GY;
291  }
293  cusparse::Multiply(cusparse_handle,cusparse_descr,&one,GH,pGX,&zero,pGY);
294  cublas::VectorTransform::dot(cublas_handle,DD,pGX,pGY,&Etemp);
295  Etemp=Etemp;
296  //Here we substitute either by the minimal or maximal bound
297  if( Etemp.real()/fabs(Etemp.real()) >0 ) if(Emax < Etemp.real())Emax=Etemp.real();
298  if( Etemp.real()/fabs(Etemp.real()) <0 ) if(Emin > Etemp.real())Emin=Etemp.real();
